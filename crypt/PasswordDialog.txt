  private void okButtonActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_okButtonActionPerformed
// Add your handling code here:
     if(keyField.getText().equals(cKeyField.getText())){
        char[] sChars = keyField.getText().toCharArray();
        for (int i = 0; i < sChars.length; i++)
        {
          if (!Character.isLetterOrDigit(sChars[i]))
          {
              md.setDialogLabel("Illegal characters in password, use letters or digits only ");
              md.show();
              return;
          }
              KSHCoder.setKeyString((new String(keyField.getPassword()).trim()));
              System.out.println("Key From password is "+keyField.getPassword());
              closeDialog(new WindowEvent(this,WindowEvent.WINDOW_CLOSING));  
          
        }
        
      }
      else{
           md.setDialogLabel("The Two Passwords Must Be The Same ");
           md.show();
           keyField.setText(null);
           cKeyField.setText(null);
      }
     
                
  }//GEN-LAST:event_okButtonActionPerformed
